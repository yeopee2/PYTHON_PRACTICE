shopping_dict = {}
for items in shopping_list :
    for item in items :
        if item not in shopping_dict.keys() :
            shopping_dict[item] = []
for key in shopping_dict.keys() :
    for items in shopping_list :
        if key in items :
            for item in items:
                if item != key :
                    shopping_dict[key].append(item)
